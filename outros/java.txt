#Configurar Links do Java
alternatives --install /usr/bin/java java /opt/java/jdk1.6.0_25/jre/bin/java 20000
alternatives --install /usr/bin/javaws javaws /opt/java/jdk1.6.0_25/jre/bin/javaws 20000
alternatives --install /usr/bin/javac javac /opt/java/jdk1.6.0_25/bin/javac 20000


#Java Browser (Mozilla) Plugin 64-bit
alternatives --install /usr/lib64/mozilla/plugins/libjavaplugin.so libjavaplugin.so.x86_64 /opt/java/jdk1.6.0_25/jre/lib/amd64/libnpjp2.so 20000

#Swap between OpenJDK and Sun/Oracle Java JDK/JRE 6 and 7 versions
alternatives --config java
alternatives --config javac
alternatives --config javaws
alternatives --config libjavaplugin.so


#export JAVA_HOME
export JAVA_HOME="/opt/java/jdk1.6.0_25"
